            methods: {
                startTyping() {
                    const type = () => {
                        const currentText = this.typingTexts[this.currentTextIndex];
                        
                        if (this.typingIndex < currentText.length) {
                            this.typingText += currentText.charAt(this.typingIndex);
                            this.typingIndex++;
                            setTimeout(type, this.typingSpeed);
                        } else {
                            setTimeout(() => {
                                this.deleteText();
                            }, 2000);
                        }
                    };
                    
                    type();
                },

                    if (this.typingText.length > 0) {
                        this.typingText = this.typingText.slice(0, -1);
                        setTimeout(() => this.deleteText(), 50);
